115 void BurstyApp::processTimer(cMessage *msg)
116 {
117  simtime_t d;
118  FSM_Switch(fsm) {
119  case FSM_Exit(INIT):
120  // transition to SLEEP state
121  FSM_Goto(fsm, SLEEP);
122  break;
123 
124  case FSM_Enter(SLEEP):
125  // schedule end of sleep period (start of next burst)
126  d = sleepTime->doubleValue();
127  scheduleAt(simTime() + d, startStopBurst);
128 
129  // display message, restore normal icon color
130  EV << "sleeping for " << d << "s\n";
131  bubble("burst ended, sleeping");
132  getDisplayString().setTagArg("i", 1, "");
133  break;
134 
135  case FSM_Exit(SLEEP):
136  // schedule end of this burst
137  d = burstTime->doubleValue();
138  scheduleAt(simTime() + d, startStopBurst);
139 
140  // display message, turn icon yellow
141  EV << "starting burst of duration " << d << "s\n";
142  bubble("burst started");
143  getDisplayString().setTagArg("i", 1, "yellow");
144 
145  // transition to ACTIVE state:
146  if (msg != startStopBurst)
147  throw cRuntimeError("invalid event in state ACTIVE");
148  FSM_Goto(fsm, ACTIVE);
149  break;
150 
151  case FSM_Enter(ACTIVE):
152  // schedule next sending
153  d = sendIATime->doubleValue();
154  EV << "next sending in " << d << "s\n";
155  scheduleAt(simTime() + d, sendMessage);
156  break;
157 
158  case FSM_Exit(ACTIVE):
159  // transition to either SEND or SLEEP
160  if (msg == sendMessage) {
161  FSM_Goto(fsm, SEND);
162  }
163  else if (msg == startStopBurst) {
164  cancelEvent(sendMessage);
165  FSM_Goto(fsm, SLEEP);
166  }
167  else
168  throw cRuntimeError("invalid event in state ACTIVE");
169  break;
170 
171  case FSM_Exit(SEND): {
172  // send out a packet
173  generatePacket();
174 
175  // return to ACTIVE
176  FSM_Goto(fsm, ACTIVE);
177  break;
178  }
179  }
180 }

182 void BurstyApp::generatePacket()
183 {
184  // generate and send out a packet
185  int destAddress = destAddresses[intuniform(0, destAddresses.size()-1)];
186 
187  char pkname[40];
188  sprintf(pkname, "pk-%d-to-%d-#%d", myAddress, destAddress, pkCounter++);
189  EV << "generating packet " << pkname << endl;
190 
191  Packet *pk = new Packet(pkname);
192  pk->setByteLength(packetLengthBytes->intValue());
193  pk->setSrcAddr(myAddress);
194  pk->setDestAddr(destAddress);
195  send(pk, "out");
196 }

198 void BurstyApp::processPacket(Packet *pk)
199 {
200  // update statistics and delete message
201  EV << "received packet " << pk->getName() << " after " << pk->getHopCount() << "hops" << endl;
202  emit(endToEndDelaySignal, simTime() - pk->getCreationTime());
203  emit(hopCountSignal, pk->getHopCount());
204  emit(sourceAddressSignal, pk->getSrcAddr());
205  numReceived++;
206  delete pk;
207 }
