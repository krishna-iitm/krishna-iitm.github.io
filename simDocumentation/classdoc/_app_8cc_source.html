158 void App::handleMessage(cMessage *msg)
159 {
160  if (msg == generatePacket)
161  {
162  if (timeSlotCounter<TotaltimeSlot)
163  {
164  trafficType::iterator it = setTrafficType.find(traffic_class_index);
165  string traffic_Class = (*it).second;
166  EV<< "Selected Traffic class is: "<<traffic_Class<<endl;
167  PacketArrivalPerSlot = poisson(lambda);//poisson(lambda);2;
168 
169  if(strcmp(traffic_Class.c_str(), "Unicast")==0)
170  {
171  EV<<"Number of unicast packets generated in the time slot: "<<timeSlotCounter<< " are "<<PacketArrivalPerSlot<<endl;
172  generateUnicastPackt(traffic_Class,PacketArrivalPerSlot);
173  }
174 
175  if(strcmp(traffic_Class.c_str(), "Broadcast")==0)
176  {
177  EV<<"Number of broadcast packets generated in the time slot: "<<timeSlotCounter<< " are "<<PacketArrivalPerSlot<<endl;
178  generateBroadcastPacket(traffic_Class,PacketArrivalPerSlot);
179  }
180 
181  if(strcmp(traffic_Class.c_str(), "Mixed")==0) // broadCast traffic generation
182  {
183  EV<<"Number of mixed (unicast plus broadcast) packets generated in the time slot: "<<timeSlotCounter<< " are "<<PacketArrivalPerSlot<<endl;
184  int num_of_unicast_pckt = intuniform(0,PacketArrivalPerSlot);
185  EV<<"Among "<<PacketArrivalPerSlot<<" packets, number of unicast packets are: "<<num_of_unicast_pckt<<endl;
186  int num_of_broadcast_pckt = PacketArrivalPerSlot - num_of_unicast_pckt;
187  EV<<"and number of broadcast packets are: "<<num_of_broadcast_pckt<<endl;
188  generateUnicastPackt(traffic_Class,num_of_unicast_pckt);
189  generateBroadcastPacket(traffic_Class,num_of_broadcast_pckt);
190  }
191 
192  timeSlotCounter++;

194  }
195  }
196  // Handle incoming packet

205  emit(totalDelayByQueue_Y_Signal, pk->getTotalQueueDelay_Y());
